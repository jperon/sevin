\couplet{
	Tandis que dort le clair de lune
	
	Sur le gazon,
	
	Rêvez, sous votre tente brune,
	
	À la maison ;
	
	Et que l'image de vos mères,
	
	Si douce à vos yeux,
	
	Emporte au pays des chimères
	
	Vos cœurs pieux.
}

\couplet{
	Autour du camp la sentinelle,
	
	Bâton en main,
	
	Poursuit sa ronde fraternelle
	
	Jusqu'à demain.
	
	Avec sa cape et sa houlette
	
	Le scoutmasteur
	
	Prend la divine silhouette
	
	Du Bon Pasteur !
	
	\smallskip
	
	\emph{Au refrain :} Car Jésus lui-même veille au bois !
}
