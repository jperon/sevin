\couplet{
	Messire le Roy, ci voyez vos \emph{Gazelles},\\
	\emph{Vos Gazelles !}\\
	Toujours les verrez rapides et fidèles\\
	Accourir au mandement du Roy.\\
	\emph{Recevez notre foi}.\\
	Sire le Roy ! nonobstant dangers\\
	Serons vos messagers,\\
	Et courrons au mandement du Roy.
}


\couplet{
	Messire le Roy, ci voyez vos \emph{Abeilles},\\
	\emph{Vos Abeilles} !\\
	Qui dessus prairie et gentes fleurs vermeilles\\
	Sont venues butiner pour le Roy.\\
	\emph{Recevez notre foi}.\\
	Sire le Roy ! sur vos grands lys francs,\\
	Royaux et conquérants,\\
	Nous venons travailler pour le Roy.
}


\couplet{
	Messire le Roy, ci voyez vos \emph{Chamois},\\
	\emph{Vos Chamois} !\\
	Guetteurs vigilants, malgré vent, neige et froid,\\
	Nous montons la garde pour le Roy.\\
	\emph{Recevez notre foi}.\\
	Sire le Roy ! sur les hauts sommets\\
	Surveillant Mahomet,\\
	Nous montons la garde pour le Roy.
}


\couplet{
	Messire le Roy, ci voyez vos grands \emph{Loups},\\
	\emph{Vos grands Loups} !\\
	Braves et rusés, à pas prudents et doux,\\
	Nous venons combattre pour le Roy.\\
	\emph{Recevez notre foi}.\\
	Sire le Roy ! à fiers coups de dents\\
	Nous lutterons ardents,\\
	Nous venons combattre pour le Roy.
}


\couplet{
	Messire le Roy, toute la \emph{Vingt-deuxième},\\
	\emph{La Vingt-deuxième} !\\
	Vous suivra, pieds nus, jusqu'à Jérusalem\\
	Où mourut le très doux Roi des Rois.\\
	\emph{Recevez notre foi}.\\
	Sire le Roy ! faut que vous nous bailliez\\
	D'être tous chevaliers,\\
	Chevaliers de Jésus Roi des Rois.
}
