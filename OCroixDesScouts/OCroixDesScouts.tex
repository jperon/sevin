\refrain{
	Ô Croix des scouts, Croix glorieuse,\\*
	Croix des héros des anciens jours,\\
	À l'âme haute et généreuse,\\*
	Tu resteras chère toujours.
}

\couplet{
	Quand il partait pour la croisade\\*
	Pour délivrer le saint Tombeau,\\
	Le chevalier, fier sans bravade,\\*
	Avait la croix pour son drapeau.
}

\couplet{
	Enrôlés sous cette bannière,\\*
	Il affrontait tous les dangers ;\\
	Il était doux à la misère,\\*
	Et courtois pour les étrangers.
}

\couplet{
	Guidé par elle, il savait être\\*
	Aimable et franc, loyal et fort,\\
	Et, sur les pas du divin Maître,\\*
	Obéissant jusqu’à la mort.
}

\couplet{
	Une croisade nous appelle\\*
	Au secours de la Vérité ;\\
	Un scout doit combattre pour elle\\*
	Toujours prêt et jamais lassé.
}

\couplet{
	Nous avons la croix pour insigne ;\\*
	Elle indique le seul vrai but,\\*
	N’oublions pas qu’elle est le signe\\
	De la victoire et du salut.
}
