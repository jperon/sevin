\couplet{
	Appris de la nature\\
	Le charme et les secrets ;\\
	Elle se transfigure,\\
	À la voir de si près.
}

\couplet{
	Goûtai la joie immense\\
	De me faire petit,\\
	Avec, pour récompense,\\
	De vivre sans souci.
}

\couplet{
	Y vis comment on s'aime\\
	Dans la paix du Seigneur,\\
	Tant que le ciel lui-même\\
	N'a pas plus de douceur.
}

\couplet{
	Près de l'autel de pierre,\\
	Le soir autour du feu,\\
	Ai compris la prière\\
	Et l'amitié de Dieu.
}

\couplet{
	Suis rentré dans la vie,\\
	D'un songe m'éveillant,\\
	L'âme pleine et ravie\\
	Et le cœur plus vaillant !
}

\couplet{
	Le plus beau jour se sauve ;\\
	Reste son souvenir :\\
	Le joli foulard mauve,\\
	Badge et lacet de cuir !
}
