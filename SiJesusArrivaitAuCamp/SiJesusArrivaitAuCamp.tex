\couplet{
	Mais bien vite on romprait les rangs,\\
	Tous ensemble lui feraient fête,\\
	Mais bien vite on romprait les rangs,\\
	Pour causer fraternellement.
}

\couplet{
	On jouerait − naturellement ! −\\
	Tous voudraient l'avoir à leur tête !\\
	On jouerait, naturellement ;\\
	On ferait qu'il gagne tout l'temps…
}

\couplet{
	Cuisiniers, pour ses aliments,\\
	Préparez fines omelettes,\\
	Cuisiniers, pour ses aliments,\\
	Soignez bien le Divin Enfant !
}

\couplet{
	Dans la paix du jour qui descend,\\
	Musiciens, sur vos clarinettes,\\
	Dans la paix du jour qui descend,\\
	Chantez tous l'Enfant-Dieu vivant !
}

\couplet{
	Aux lueurs du bon feu de camp,\\
	Avec sa Sagesse parfaite,\\
	Aux lueurs du bon feu de camp,\\
	Il nous parle divinement !
}

\couplet{
	D'écouter ses enseignements,\\
	Bien des pleurs à couler s'apprêtent,\\
	D'écouter ses enseignements,\\
	Chacun pense à son grand serment.
}

\couplet{
	Pour la nuit, comme logement,\\
	Tous voulant offrir leurs couchettes,\\
	Il s'en va prendre logement,\\
	Sous la tente du plus fervent…
}

\couplet{
	Or Jésus est toujours au camp ;\\
	Pas besoin de nous mettre en quête,\\
	Or Jésus est toujours au camp,\\
	Invisible et partout présent.
}

\couplet{
	Dieu est là, dans les fleurs des champs,\\
	Les nuages et les planètes,\\
	Dieu est là, dans les fleurs des champs,\\
	Dans nos frères, petits et grands.
}

\couplet{
	Il est là, dans le chef du camp ;\\
	C'est sa voix qui parle, discrète.\\
	Il est là, dans le chef du camp ;\\
	Soyons-lui bien obéissants !
}

\couplet{
	Il est là dans les mendiants\\
	Qui devant nos repas s'arrêtent\\
	Il est là dans les mendiants :\\
	Donnons-lui généreusement.
}

\couplet{
	Quand arrive quelque accident,\\
	C'est Jésus, le blessé qu'on traite,\\
	Quand arrive quelque accident,\\
	C'est Jésus qu'on panse vraiment !
}

\couplet{
	Ô Jésus, venez donc au camp :\\
	Pour la messe la Table est prête,\\
	Ô Jésus, venez donc au camp,\\
	Donnez-nous votre Sacrement.
}

\couplet{
	Ô Jésus, demeurez au camp !\\
	Rendez-y nos âmes parfaites,\\
	Pour qu'un jour, tous, dans votre camp\\
	Nous campions éternellement !
}
