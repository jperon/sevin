\couplet{
	Par ton départ de Vaucouleurs,\\*
	Par Orléans libre et vainqueur,\\
	Garde en nos cœurs jusqu'à la mort\\*
	L'\emph{Espoir} sans cesse jeune et fort.
}

\couplet{
	Par le « Service » le plus haut,\\*
	Par le martyre le plus beau,\\
	Par le cri de ton dernier jour,\\*
	Garde-nous, Jeanne, dans l'\emph{Amour}.
}
