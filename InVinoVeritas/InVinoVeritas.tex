\couplet{
	\emph{In vino veritas}, mes frères,\\*
	Dit un vieux proverbe latin :
}

\couplet{
	Dieu, pour nous faire aimer nos verres,\\*
	Mit la vérité dans le vin !
}

\couplet{
	Le bon vin est un bien suprême,\\*
	D’un vieux buveur voici l’avis :
}

\couplet{
	Ce n’est pas que le vin que j’aime,\\*
	Mes amis ; c’est la vérité !
}
