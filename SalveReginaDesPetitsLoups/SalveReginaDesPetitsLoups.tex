\couplet{
	Que par vos soins, Mère de Grâce,\\
	De notre cœur Jésus efface\\
	De nos fautes, de nos méfaits\\
	La trace,\\
	Pour ne l'offenser désormais\\
	Jamais.
}

\couplet{
	Accordez-nous d'être bien sages,\\
	De croître en grâce comme en âge,\\
	Et d'être, ô Reine des Vertus,\\
	Vos pages,\\
	Afin d'aimer de plus en plus\\
	Jésus.
}

\couplet{
	Au dernier jour de notre vie,\\
	Entre vos bras, Mère bénie,\\
	Au ciel que rêve notre espoir,\\
	Marie,\\
	Daignez enfin nous recevoir :\\
	Bonsoir !
}
