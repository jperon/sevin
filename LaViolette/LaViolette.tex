\couplet{
	J’ai une amie dans le monde,\\
	Mais elle est bien loin hélas !%
	\raisebox{1.5ex}[0pt][0pt]{\hspace{3.7em}\} \emph{bis}\hspace{-3.7em}}\\
	J’ai quelque chose à lui dire,\\
	Oui mais qui le lui dira ?	
}

\refrain{
	La violette double, double,\\
	La violette doublera.%
	\raisebox{1.5ex}[0pt][0pt]{\hspace{3.7em}\} \emph{bis}\hspace{-3.7em}}
}

\couplet{
	Si je l’dis à l’alouette,\\
	Tout le monde le saura\\
	Rossignol du ver bocage,\\
	Faites-moi ce plaisir-là !	
}

\couplet{
	Rossignol prend sa volée\\
	Au château d’amour s’en va ;\\
	Trouvant la porte fermée, \\
	Par la fenêtre il entra.	
}

\couplet{
	Bonjour l’une, bonjour l’autre,\\
	Bonjour, la bell' que voilà !\\
	C’est votre amant qui demande\\
	Que vous ne l’oubliiez pas.	
}

\couplet{
	Quoi, mon amant me demande\\
	Que ne l’oublie pas ?\\
	J’en ai oublié bien d’autres\\
	J’oublierai bien celui là.	
}

\couplet{
	S’il était venu lui-même\\
	N’aurait pas perdu ses pas\\
	Tout amant qui craint sa peine\\
	Restera dans l’embarras.
}
