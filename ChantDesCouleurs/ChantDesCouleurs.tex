\refrain{
	Chevaliers, saluons les couleurs,\\*
	Sonne, sonne éclaireur, sonne les honneurs.\\*
	Sonne bien, sonne de tout ton cœur !\\*
	Sonne, sonne éclaireur, sonne les honneurs.\par
}

\strophe{
	Pour nous c’est fête,\\*
	Quand, sur nos têtes,\\*
	Notre drapeau\\*
	Flotte bien haut.\\
	Quand viendra l’ombre\\*
	Et la nuit sombre,\\*
	Ses plis sacrés\\*
	Seront repliés.\par
}
