%Compilation:cd .. && ./chants-scouts.sh
\couplet{
	Donc en route, et que sonne la campagne !\\
	Clair et vif, le pipeau nous accompage :\\
	Faut-il des difficultés\\
	Pour qu'un scout aime à chanter ?
}


\couplet{
	Que la pluie sur nos tentes tambourine,\\
	Que le vent s'engouffre en nos pèlerines,\\
	Nos refrains plus éclatants\\
	Feront fuir le mauvais temps.
}


\couplet{
	Quand, méfiant devant notre allure étrange,\\
	On nous dit : \og Non, vous n'aurez pas ma grange ! \fg\\
	Nous partons, si réjouis,\\
	Les gens croient avoir dit : \og Oui ! \fg
}


\couplet{
	Déridons les pleurards et les moroses !\\
	Quand on veut, les épines ont des roses :\\
	Nos ennuis, nous les chassons\\
	Á coups d'aile de chansons.
}


\couplet{
	Nous chantons, ma foi Dieu, pour être apôtres :\\
	Être scout, c'est penser d'abord aux autres.\\
	Á tous ceux dont le cœur ploie\\
	Nous voulons porter la joie.
}


\couplet{
	Nous chantons, car pour nous la vie est belle :\\
	Nous croyons : notre joie est immortelle !\\
	Dieu nous fit pour être heureux\\
	Et nous garde encor bien mieux.
}


\couplet{
	Nous chantons, la tristesse use et divise ;\\
	Nous chantons pour tenir notre devise :\\
	Tout aimer, ne rien haïr,\\
	Et surtout ne pas vieillir !
}


\couplet{
	Lorsqu'un jour, notre tâche bien finie,\\
	L'appel scout viendra clore notre vie,\\
	Au ciel où Dieu nous attend\\
	Nous irons tout en chantant.
}
