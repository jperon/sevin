\couplet{
	Pas besoin d'énormes troupes
	
	Où l'on est cent quatre-vingts :
	
	Nous cherchons les petits groupes
	
	Où l'on s'aime en vieux copains !
	
	\textit{C'est nos pat', nos pat', nos pat',}
	
	\textit{C'est nos patrouilles qu'il nous faut !}
}


\couplet{
	Pas besoin de vestes chiques,
	
	De cravates, de calots,
	
	Nous trouvons bien plus pratiques
	
	Nos foulards et nos chapeaux !
	
	\textit{C'est la ch'mis', la ch'mis', la ch'mis',}
	
	\textit{C'est la ch'mise qu'il nous faut !}
}


\couplet{
	Pas besoin, faux militaires,
	
	De singer les officiers,
	
	C'est bon pour les \og scoutmastères \fg
	
	Qui sabotent leur métier !
	
	\textit{C'est des scouts, des scouts, des scouts,}
	
	\textit{Des scoutmestres qu'il nous faut !}
}


\couplet{
	Pas besoin d'fumer des pipes
	
	Aux terrasses des cafés,
	
	C'est bon pour les pauvres types
	
	Qui n'sav'nt pas comment s'chauffer !
	
	\textit{C'est la flamm', la flamm', la flamm',}
	
	\textit{La flamm' du camp qu'il nous faut !}
}


\couplet{
	Pas besoin non plus d'\og écorces \fg
	
	Des journalistes V. P. :
	
	Suis ta piste, apprends le Morse,
	
	Sauveteur, sois \emph{toujours prêt} :
	
	\textit{C'est la nag', la nag', la nag',}
	
	\textit{C'est la nage qu'il nous faut !}
}


\couplet{
	Pas besoin qu'on nous disserte
	
	De lois sur les logements,
	
	C'est dans la campagne verte
	
	Que sont nos appartements.
	
	\textit{C'est la tent', la tent', la tent',}
	
	\textit{C'est la tente qu'il nous faut !}
}


\couplet{
	Pas besoin de confortable,
	
	Et, lorsqu'on s'en va camper,
	
	D'aboir son tub et sa table,
	
	Á quand donc les canapés !
	
	\textit{C'est la paill', la paill', la paill',}
	
	\textit{C'est la paille qu'il nous faut !}
}


\couplet{
	Pas besoin d'entrer en ville
	
	Tout en faisant du raffût :
	
	Le vrai scout, pour être utile,
	
	Doit tout voir, non être vu !
	
	\textit{C'est le trav', le trav', le trav',}
	
	\textit{C'est le travail qu'il nous faut !}
}


\couplet{
	Pas besoin donc de ces fêtes
	
	Où l'on défile, alignés,
	
	Encadrés par les trompettes,
	
	Les rosièr's et les pompiers !
	
	\textit{C'est la brouss', la brouss', la brouss',}
	
	\textit{C'est la brousse qu'il nous faut !}
}


\couplet{
	Pas besoin d'grands politiques
	
	Pour venir nous haranguer :
	
	Qu'ils demeur'nt dans leurs boutiques,
	
	Notre chef est bien plus gai !
	
	\textit{C'est le chef, le chef, le chef,}
	
	\textit{C'est notre chef qu'il nous faut !}
}


\couplet{
	Pas besoin d'avoir le nombre,
	
	Ce qu'il faut, c'est la valeur :
	
	Des zéros qui nous encombrent
	
	Débarrassez-nous, Seigneur !
	
	\textit{C'est des scouts, des scouts, des scouts,}
	
	\textit{C'est des vrais scouts qu'il nous faut !}
}
