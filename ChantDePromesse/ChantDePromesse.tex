\couplet{
	Je jure de te suivre\\
	En fier chrétien,\\
	Et tout entier je livre\\
	Mon cœur au tien.
}

\couplet{
	Fidèle à ma Patrie,\\
	Je le serai ;\\
	Tous les jours de ma vie,\\
	Je servirai.
}

\couplet{
	Je suis de tes apôtres,\\
	Et chaque jour\\
	Je veux aider les autres\\
	Pour ton amour.
}

\couplet{
	Ta Règle a sur nous-mêmes\\
	Un droit sacré ;\\
	Je suis faible, tu m'aimes,\\
	Je maintiendrai !
}
