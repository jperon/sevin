\couplet{
	Je circulais donc -- pour tout voir --
	
	Lorsque soudain, entre les branches,
	
	Aiguisant déjà ses dents blanches,
	
	Surgit un énorme loup noir :
	
	Et je voyais dans la nuit brune
	
	Le corps du monstre frétiller,
	
	Et ses yeux effrayants briller
	
	\emph{Au clair de la lune !}
}

\couplet{
	Alors, simplement par devoir,
	
	Je prends mon sifflet et j'appelle :
	
	Chacun prend sa pioche ou sa pelle…
	
	Que la bravoure est belle à voir !
	
	Mais, malechance peu commune,
	
	Jamais le loup ne fut trouvé ;
	
	-- Ils ont dit que j'avais rêvé
	
	\emph{Au clair de la lune !}
}
