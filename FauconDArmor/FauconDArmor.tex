\couplet{
  Ses yeux fixent l'Océan glauque
  
  Et le soleil,
  
  Sa gorge lance son cri rauque,
  
  Strident réveil.
}


\couplet{
  Quand il se sent forte la serre
  
  Et le bec fort,
  
  Impétueux comme un corsaire,
  
  Prend son essor.
}


\couplet{
  Chinois à la face sournoise
  
  Ont peur de lui,
  
  Et dans la forêt tonkinoise
  
  Le tigre a fui.
}


\couplet{
  \mbox{Dans notre grande île africaine,}
  
  L'œil vigilant,
  
  Monte la garde, fier, sans haine,
  
  Étincelant.
}


\couplet{
  Quand sonne la chanson des balles
  
  S'envole au front,
  
  Et dans ses serres triomphales
  
  Prend Douaumont.
}


\couplet{
  Maintenant sur toute la France
  
  Il lance encor
  
  Son cri de veille et d'espérance,
  
  Faucon d'Armor.
}


\couplet{
  Vieux guetteur toujours à l'écoute,
  
  Il sait tout voir,
  
  Et son vol nous trace la route
  
  Vers le devoir.
}


\couplet{

  Pour le gravir,
