\couplet{
	Adieu, repas
	
	Qu'on prenait sur le pouce,
	
	Et cuisiniers
	
	Qui nous empoisonniez :
	
	Nous n'irons pas
	
	Déjeuner sur la mousse
	
	En r'gardant l'horizon.
}

\couplet{
	Adieu, grand bois
	
	Qui vis nos jeux rustiques,
	
	Où les oiseaux
	
	Chantaient dans les berceaux,
	
	Où notre voix
	
	Fit monter ses cantiques
	
	En joyeuse oraison.
}

\couplet{
	Adieu, bon feu
	
	Des veillées fraternelles,
	
	Qui flamboyais
	
	Et qui nous égayais ;
	
	Et vous, adieu,
	
	Rondes des sentinelles ;
	
	Changeons de garnison.
}
