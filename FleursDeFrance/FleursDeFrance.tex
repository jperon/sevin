\couplet{
	C'est un' Reine de France,
	
	Fleur de lys, fleur d'orange,
	
	Qui en Clovis nous baptisa,
	
	Fleur de lys, fleur de lilas.
}


\couplet{
	C'est un' Bergèr' de France,
	
	Fleur de lys, fleur d'orange,
	
	Qui sauva Paris d'Attila :
	
	Y en a pas deux comm' cell'-là.
}

}

}


\couplet{

}


\couplet{

}


\couplet{

	Fleur de lys, fleur d'orange,

}
