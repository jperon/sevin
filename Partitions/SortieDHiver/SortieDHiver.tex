\couplet{%
	Nous arrivons le nez bleu,\\*
	Les jambes violettes,\\
	Sentant encore dans nos yeux\\*
	L'air de la bicyclette!
}

\couplet{%
	À la fin de la veillée,\\*
	Serrés sous la tente,\\
	Nous restons tout habillés,\\*
	L'échine frissonnante !
}

\couplet{%
	Couchés, le froid nous poursuit :\\*
	Pas le moindre somme ;\\
	Vaut bien mieux passer la nuit\\*
	À se fendre la pomme !
}

\couplet{%
	Au moment de se laver,\\*
	Le CP sévère,\\
	Ne nous permet de garder\\*
	Qu'un slip sur le derrière !
}

\couplet{%
	Je rentre me cuire la peau\\*
	Dans l’eau d'la baignoire ;\\
	M'enfiler du bouillon chaud\\*
	Et des suppositoires !
}

\couplet{%
	Adieu, pour moi c’est trop dur,\\*
	À bas le scoutisme !\\
	J'irai sur la côte d'azur,\\*
	L'été, faire du campisme !
}
