\couplet{%
	Mondes que souleva\\*
	La main de Jéhovah,\\
	Bénissez Marie ;\\
	Houles des océans\\
	Aux abîmes béants,\\*
	Bénissez Marie.
}

\couplet{%
	Brouillards bleus des matins,\\*
	Estompant les lointains,\\
	Bénissez Marie ;\\
	Couchants de pourpre et d'or,\\
	Plus splendides encor,\\*
	Bénissez Marie.
}

\couplet{%
	Ô printemps vert et doux,\\*
	Et soleil des mois d'août,\\
	Bénissez Marie ;\\
	Automne au front couvert,\\
	Neige des camps d'hiver,\\*
	Bénissez Marie.
}

\couplet{%
	Plaines qui déroulez\\*
	Le tapis d'or des blés,\\
	Bénissez Marie ;\\
	Montagnes où sans peurs\\
	S'élancent nos grimpeurs,\\*
	Bénissez Marie.
}

\couplet{%
	Torrents qui vous cachez\\*
	De rochers en rochers,\\
	Bénissez Marie ;\\
	Rivières au flot lent,\\
	Tout en vous écoulant,\\*
	Bénissez Marie.
}

\couplet{%
	Sentiers dont les buissons\\*
	Sont remplis de chansons,\\
	Bénissez Marie ;\\
	Chaleur des grands chemins\\
	Et fraîcheur des ravins,\\*
	Bénissez Marie.
}

\couplet{%
	Chênes de la forêt\\*
	Où le jour disparaît,\\
	Bénissez Marie ;\\
	Fleurs frêles des vallons\\
	Où nous nous installons,\\*
	Bénissez Marie.
}

\couplet{%
	Tentes de nos campeurs\\*
	Couronnant les hauteurs,\\
	Bénissez Marie ;\\
	Troncs des pins rougeoyants,\\
	Feux de camp flamboyants,\\*
	Bénissez Marie.
}

\couplet{%
	Fanions qui palpitez\\*
	Aux souffles des étés,\\
	Bénissez Marie ;\\
	Trompes du soir qui meurt,\\
	Lancez votre clameur,\\*
	Bénissez Marie.
}

\couplet{%
	Ô frères inconnus\\*
	Que l'on n'a jamais vus,\\
	Bénissez Marie ;\\
	En vos parlers divers,\\
	Scouts de tout l'univers,\\*
	Bénissez Marie.
}
