\couplet{%
	Comme un enfant,\\
	Ignorant le blasphème,\\
	Je veux que mon cœur aime.\\
	Comme un enfant,\\
	Je voudrais, si je pleure,\\
	Qu'en moi l'espoir demeure…\\
	Comme un enfant.
}

\couplet{%
	Comme un enfant,\\
	À tout prix je veux croire,\\
	Seigneur, en ta victoire !\\
	Comme un enfant,\\
	Si la vie est méchante,\\
	Cache-moi sous ta mante\\
	Comme un enfant.
}

\couplet{%
	Comme un enfant,\\
	En ton Eucharistie,\\
	Je veux puiser la vie ;\\
	Comme un enfant,\\
	Si pourtant je chancelle,\\
	Fais que je sois fidèle,\\
	Comme un enfant !
}

\couplet{%
	Comme un enfant,\\
	Quand les dernières fièvres\\
	Auront brûlé mes lèvres,\\
	Comme un enfant,\\
	Seigneur, qu'avec les Anges\\
	Je chante tes louanges\\
	Comme un enfant.
}
