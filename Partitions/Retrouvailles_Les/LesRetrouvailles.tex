\couplet{%
	Le temps est loin de nos vingt ans,\\*
	Des coups de poing, des coups de sang,\\
	Mais qu’a c’la n’tienne, c’est pas fini,\\*
	On peut chanter quand le verre est bien rempli.
}

\refrain{
	Buvons encore, une dernière fois,\\*
	À l’amitié, l’amour, la joie,\\
	On a fêté nos retrouvailles ;\\
	Ça m’fait d’la peine,\\*
	Mais il faut que je m’en aille.
}

\couplet{%
	Je suis parti changer d’étoile,\\*
	Sur un navire j’ai mis la voile,\\
	Pour n’être plus qu’un étranger\\*
	Ne sachant pas très bien où il allait.
}

\couplet{%
	Mais souviens-toi de cet été,\\*
	La première fois qu’on s’est saoûlé :\\
	Tu m’as ram'né à la maison,\\*
	En chantant, on marchait à reculons.
}

\couplet{%
	J't’ai raconté mon mariage,\\*
	À la mairie d’un p'tit village,\\
	Je rigolais dans mon plastron,\\*
	Quand le maire essayait d'prononcer mon nom.
}

\couplet{%
	J’n'ai pas écrit toutes ces années,\\*
	Et toi aussi t’es marié :\\
	T’as trois enfants à faire manger,\\*
	Mais j’en ai cinq, si ça peut te consoler.
}
