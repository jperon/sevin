\couplet{%
  Voyez, ô Notre-Dame,\\*
  Vos enfants devant vous.\\
  Ils viennent vous prier,\\
  \emph{Gratia plena} :\\
  Vous les écouterez,\\*
  \emph{Mater alma}.
}

\couplet{%
  Ils voudraient que leur âme\\*
  Soit toujours belle pour vous,\\
  Et leur vie à vos yeux,\\
  \emph{Gratia plena},\\
  Comme un reflet de Dieu,\\*
  \emph{Mater alma}.
}

\couplet{%
  Ô vous, l'amour immense,\\*
  Rendez leur cœur plus grand\\
  Et d'amour débordant,\\
  \emph{Gratia plena},\\
  Puis élevez-le vers vous,\\*
  \emph{Mater alma}.
}

\couplet{%
  Accordez-leur la grâce,\\*
  Leur service achevé,\\
  D'accueillir ce moment,\\
  \emph{Gratia plena},\\
  Sans peur, sereinement,\\*
  \emph{Mater alma}.
}
