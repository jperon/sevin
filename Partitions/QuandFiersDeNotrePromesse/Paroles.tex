\couplet{
	Quand, fiers de notre promesse,\\*
	Nous élevons la main \emph{ého},\\
	Doigts joins, cœur sans faiblesse,\\*
	Sans crainte du lendemain \emph{ého}.
}

\refrain{
	Tralala ého, tralala ého !\\*
	Vivent les Scouts, ého.
}

\couplet{
	Tout chante en nous l'allégresse\\*
	De servir et d'aimer,\\
	Et d'être prêts sans cesse\\*
	Pour tous à nous dévouer.
}

\couplet{
	La joie est notre partage,\\*
	Légers et francs et purs,\\
	Tel un oiseau sans cage\\*
	Chantant, libre dans l'azur.
}
