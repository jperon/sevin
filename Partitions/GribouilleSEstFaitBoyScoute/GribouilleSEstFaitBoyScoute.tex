\couplet{
	Le v'là qui s'met en route\\
	Pour s'en aller camper.
}

\couplet{
	Il n'a pris pour bagage…\\
	Qu'un sac de vingt kilos.
}

\couplet{
	Il s'étonn' d'être en nage\\
	Et d'en avoir… plein l'dos !
}

\couplet{
	Au premier kilomètre\\
	Gribouille traîne le pas.
}

\couplet{
	Au deuxième, il faut l'mettre\\
	Sur la charrette à bras.
}

\couplet{
	Au camp, l'ami Gribouille\\
	Veut fair' sa bonne action.
}

\couplet{
	À son chef de patrouille\\
	Il donn' des instructions.
}

\couplet{
	Il veut fair' la cuisine,\\
	Tous ses plats sont brûlés !
}

\couplet{
	On trouv' de la résine\\
	Dans son café au lait.
}

\couplet{
	Il déchire sa culotte\\
	En courant dans les bois.
}

\couplet{
	Il tombe à l'eau, se crotte,\\
	Et se perd deux, trois fois.
}

\couplet{
	Il jugeait résistante\\
	La tent' qu'il installa ;
}

\couplet{
	Pendant la nuit, la tente\\
	Sur son nez s'écroula.
}

\couplet{
	Gribouill' retourne en ville\\
	Plus vit' qu'il n'était v'nu ;
}

\couplet{
	En disant, l'imbécile :\\
	« On ne m'y prendra plus ! »
}

\couplet{
	« Si c'est ça l'camp, j'y coupe,\\
	Malad' je m'fais porter ! »
}

\couplet{
	(Gribouill' n'est pas d'notr' troupe,\\
	Mais d'la troup' d'à côté !)
}
