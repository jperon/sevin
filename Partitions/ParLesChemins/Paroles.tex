\couplet{%
	Par les chemins joyeux où l'amour nous entraîne,\\*
	Dès le matin,\\
	Ayant pour seul trésor le lourd bâton de frêne\\*
	Dans notre main,
}

\refrain{%
	Cœur allègre,\\*
	Front intègre,\\
	Sans souci des lendemains,\\
	Tête haute,\\
	Gais sans faute,\\*
	Nous allons par les chemins.
}

\couplet{%
	Par les chemins de plaine où le soleil surplombe\\*
	Les durs chemins,\\
	Au voyageur meurtri qui chancelle et qui tombe\\*
	Tendant nos mains,
}

\couplet{%
	Par les chemins de lande où le vent bat sa ronde,\\*
	Les chemins las,\\
	À ceux qui marchent seuls, loin du ciel et du monde,\\*
	Tendant nos bras,
}

\couplet{%
	Par les chemins ardus côtoyant les abîmes,\\*
	Au flanc des monts,\\
	Les yeux tendus au ciel, nous gagnerons les cîmes,\\*
	Où nous mourrons…
}
