\couplet{%
	Avez-vous vu\\*
	Ces 6 sauvages\\*
	Qui débarquaient\\*
	Hier à Paris ?\\*
	Ils étaient noirs\\*
	Comme du cirage,\\*
	Depuis la tête\\*
	Jusqu’au nombril !
}

\refrain{
	Je cherche fortune\\*
	Tout au long du Chat Noir ;\\*
	Et au clair de la lune,\\*
	À Montmartre le soir.
}
