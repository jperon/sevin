\couplet{%
	\emph{Le gamin :}\\*
	Où vas-tu, petit Scout de France,\\
	Où vas-tu d'un pas si léger ?\\
	\emph{Le scout :}\\*
	Je n'ai peur d'aucune distance,\\
	Du bonheur je suis messager :\\
	Si tu veux aimer l'existence,\\*
	Avec nous, viens la partager !
}

\couplet{%
	\emph{Le gamin :}\\*
	Que fais-tu le long des grand-routes,\\
	Que fais-tu, quel est donc ton sort ?\\
	\emph{Le scout :}\\*
	De mon cœur, bravement, j'écoute\\
	Le désir d'être pur et fort,\\
	Si tu veux garder la loi scoute,\\*
	Avec nous, viens aimer l'effort !
}

\couplet{%
	\emph{Le gamin :}\\*
	Diras-tu de quelle espérance\\
	Est rempli ton amour pieux ?\\
	\emph{Le scout :}\\*
	J'ai reçu de ma douce France\\
	La foi pure de mes aïeux.\\
	Si tu veux garder ta croyance,\\*
	Avec nous viens et prions Dieu !
}

\couplet{%
	\emph{Le gamin :}\\*
	Quand tu dors à la belle étoile,\\
	Ah ! dis-moi quel est ton plaisir…\\
	\emph{Le scout :}\\*
	Dans les camps, la nuit, sous la toile,\\
	Je deviens hardi pour souffrir.\\
	Généreux, le Scout se dévoile ;\\*
	Avec nous viens aussi dormir !
}
