\couplet{
	Recevez-nous, bon Saint François,\\*
	Pour vos très humbles petits frères.\\
	Deux sous pourraient couvrir nos terres :\\*
	Nous pouvons vivre sous vos lois ;\\*
	Recevez-nous, bon Saint François.
}

\couplet{
	Nous serons sages, Saint François !\\*
	Comme le loup de vos légendes,\\
	Sans gronderies ou réprimandes\\*
	Vos Petits-Loups marcheront droit !\\*
	Nous serons sages, Saint François.
}

\couplet{
	Enseignez-nous, bon Saint François,\\*
	Votre doctrine merveilleuse :\\
	Elle rend l'âme si joyeuse !\\*
	Seul le péché cause l'effroi,\\*
	Enseignez-nous, bon Saint François.
}

\couplet{
	Vous qui chantiez, bon Saint François,\\*
	Dieu rayonnant dans la nature,\\
	Conservez-nous une âme pure,\\*
	Pour mieux le voir dans les grands bois\\*
	Et le chanter, bon Saint François.
}

\couplet{
	À vos leçons, bon Saint François,\\*
	Les oiseaux même étaient dociles ;\\
	Rendez-nous donc comme eux faciles\\*
	Dès que le chef lève le doigt,\\*
	Par vos leçons, bon Saint François.
}

\couplet{
	Vous ressembler, bon Saint François,\\*
	C'est ressembler au Divin Maître ;\\*
	On vit en vous Jésus paraître\\*
	Avec les marques de sa Croix !\\*
	Vous ressembler, bon Saint François !…
}
