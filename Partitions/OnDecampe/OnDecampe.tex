\refrain{
	Oh ! la, la, misère et branle-bas !\\*
	Voilà qu’on décampe, \\*
	Oh ! la, la, misère et branle-bas !\\*
	Voilà qu’on s’en va.
}

\couplet{%
	Sur l’dos, qui prendra la tente et les cordeaux,\\*
	Qui prendra la pelle \emph{et les marteaux},\\*
	\emph{Et les marteaux} ? \vbis
}

\couplet{%
	Sur l’dos, qui prend la marmite Bonnamaux,\\*
	Qui prend les piquets \emph{et les lassos} ?
}

\couplet{%
	Sur l’dos, qui prend la bouillotte et le réchaud,\\*
	Qui prend la cuvette \emph{et le vieux seau} ?
}

\couplet{%
	Sur l’dos, qui prend la moutarde et les pruneaux,\\*
	Qui prend la passoire \emph{et les poireaux} ?
}
