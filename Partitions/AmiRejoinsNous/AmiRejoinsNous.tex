\refrain{
	Ami, rejoins-nous sur le chemin,\\*
	Portant ton fardeau avec entrain ;\\
	Quitte la pauvreté du confort\\*
	Reçois les richesses de l'effort !\\
	Marchons, en avant marchons. \vbis
}

\couplet{
	Quand tu crois que tu es seul au monde,\\*
	Isolé dans la nuit sans secours,\\
	Apporte-nous ta charge,\\*
	Nous la partagerons. 
}

\couplet{
	Quand tu souffres en ton corps, en ton âme,\\*
	Redoutant les embûches du chemin,\\
	Reprends, reprends courage,\\*
	Le Christ est devant toi. 
}

\couplet{
	Si tu suis ceux qui ouvrent la marche,\\*
	Tu soutiens ceux qui suivent ton pas,\\
	Nous sommes solidaires,\\*
	À la vie, à la mort. 
}

\couplet{
	La tempête et le froid et l'orage,\\*
	Le soleil et la soif et la faim,\\
	Ne pourront pas réduire\\*
	L'allure de ton pas. 
}

\couplet{
	Souviens-toi que la croix, la souffrance\\*
	Acceptées dans la Foi, par amour,\\
	Seront la voie unique\\*
	Pour te mener à Dieu. 
}

\couplet{
	Parcourons campagnes et villages,\\*
	Les faubourgs et le cœur des cités,\\
	Faisons sonner ces cloches,\\*
	Cloches qui se sont tues. 
}
