\couplet{
	Devant ma fenêtre\\*
	Y’a un oisillon,\\*
	Toute la nuit chante,\\*
	Chante sa chanson.
}

\refrain{
	S’il chante, qu’il chante :\\*
	Ce n’est pas pour moi,\\*
	Mais pour mon amie\\*
	Qui est loin de moi.\\*
}

\couplet{
	Ces hautes montagnes,\\*
	Ces longues vallées\\*
	Cachent le visage \\*
	De ma bien-aimée.
}

\couplet{
	Baissez-vous montagnes,\\*
	Plaines, haussez-vous :\\*
	Que je voie sans peine \\*
	Où sont mes amours !
}
