\refrain{
	En avant, les Scouts, au soleil levant,\\*
	La grand'route est claire et sonore.\\
	Du gauche et du droit, toujours en avant,\\*
	Et chantons à l'aurore.
}

\couplet{
	Sortant des brumes du matin,\\*
	Dans la nature où tout s'éclaire,\\
	Le gai soleil paraît soudain,\\*
	Lançant la flamme et la lumière.
}

\couplet{
	Les lourds brouillards ont disparu\\*
	Aux vertes berges des rivières,\\
	Le flot que rien ne trouble plus\\*
	Renvoie au soleil sa lumière.
}

\couplet{
	Tout someillait autour de nous :\\*
	Les bois étaient pleins de mystère ;\\
	Mais, à présent, quels réveils fous !\\*
	Tout chante et rit à la lumière.
}

\couplet{
	Avec nos frères les oiseaux\\*
	Au ciel s'adressent nos prières,\\
	Scouts, Vieux Routiers et Louveteaux,\\*
	Dieu seul nous donne sa lumière.
}

\couplet{
	Aussi longtemps que le soleil\\*
	Prendra sa route sur nos terres,\\
	Tous, pleins d'amour pour nos pareils,\\*
	Nous combattrons pour la lumière !
}
