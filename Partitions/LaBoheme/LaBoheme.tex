\couplet{
	Chante et  danse la Bohême, \emph{faria, faria, ho !}\\*
	Vole et campe où Dieu la mène, \emph{faria, faria, ho !}\\*
	Sans souci au grand soleil, \\*
	Coule des jours sans pareils.\\*
	\emph{Faria, faria, faria, faria, faria, faria, ho !}
}

\couplet{
	Sur la mousse ou dans la paille,\\*
	Trouve un lit fait à sa taille.\\*
	Cœur léger Bohême dort,\\*
	Que n’éveille aucun remords.
}

\couplet{
	Quand la faim se fait tenace,\\*
	Dans les bois se met en chasse :\\*
	Tendre biche ou prompt chamois,\\*
	Lui feront un repas de roi.
}

\couplet{
	Quand la soif brûle sa gorge,\\*
	Au ravin la source est proche :\\*
	Eaux plus claires que l’Asti,\\*
	En vous tout le ciel sourit.
}

\couplet{
	Dans sa bourse rien ne pèse,\\*
	Et son cœur bat tout à l’aise :\\*
	Point de comptes, point d’impôts,\\*
	Rien ne trouble son repos.
}
