\let\vcouplet\couplet
\renewcommand{\couplet}[2]{\vcouplet{
	\textsc{#1}\\
	#2
}
}

\couplet{Shere Khan}{%
	Je n'ai point garde :\\
	Tu n'auras pas ma peau.
}

\couplet{Mowgli}{%
	Roi de la Jungle,\\
	Je viens à vos genoux.
}

}

}

	Je n'ai pas peur de vous.
}

}

}

}

}

}

}
