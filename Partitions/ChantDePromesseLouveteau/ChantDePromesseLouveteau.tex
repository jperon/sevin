\couplet{%
	À toi, Jésus que j'aime,\\*
	J'obéirai,\\*
	Mais m'écouter moi-même,\\*
	Non, non jamais.%
}

\refrain{%
	Je promets d'être sage,\\*
	Ô bon Jésus,\\*
	Fais-moi grandir en âge\\*
	Et en vertu.%
}

\couplet{%
	Penser d'abord aux autres,\\*
	Dire toujours vrai :\\*
	Ces maximes sont nôtres,\\*
	Je les suivrai.%
}

\couplet{%
	Ô Saint-François d'Assise,\\*
	Enseignez-moi,\\*
	À garder ma devise,\\*
	Et notre loi.%
}

\couplet{%
	Bon p'tit loup je veux être,\\*
	Pour devenir\\*
	Un scout, ô tendre Maître,\\*
	C'est mon désir.%
}
