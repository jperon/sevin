\couplet{%
	Qui peut faire de la voile sans vent,\\*
	Qui peut ramer sans rames ?\\
	Et qui peut quitter son ami\\*
	Sans verser une larme ?
}

\couplet{%
	Je peux faire de la voile sans vent,\\*
	Je peux ramer sans rames,\\
	Mais ne peux quitter mon ami\\*
	Sans verser une larme.
}
