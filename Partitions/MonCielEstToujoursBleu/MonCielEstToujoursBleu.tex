\couplet{
	À notre camp la vie est bonne,\\
	Aux cœurs des scouts il faut si peu,\\
	La charité partout rayonne,\\
	Chacun sourit, chacun se donne :\\
	Le ciel est toujours bleu ! \vbis
}

\couplet{
	À notre camp la vie est gaie,\\
	Le soir on chante autour du feu.\\
	Près de Jésus qui donc s'effraie ?\\
	Le rire est pur, la joie est vraie,\\
	Le ciel est toujours bleu !
}

\couplet{
	À notre camp la vie est forte\\
	Et le travail est notre jeu.\\
	S'il faut peiner, peu nous importe !\\
	Un bel effort nous réconforte :\\
	Le ciel est toujours bleu !
}

\couplet{
	À notre camp la vie est sainte,\\
	Jésus nous parle et nous émeut ;\\
	Dieu se promène en notre enceinte\\
	Et nous dormons sans nulle crainte :\\
	Le ciel est toujours bleu !
}
