\couplet{%
	Faut repeindre la charrette\\
	Qui ne peut rester comm' ça ;\\
	Je voudrais qu'elle soit chouette :\\
	Qui donc me la repeindra ?\\
	-- \emph{La Chouette,} la Chouette,\\
	La Chouette le fera.
}

\couplet{%
	Faudrait réparer la tente,\\
	Qu'un dernier camp déchira ;\\
	Si l'on prolonge l'attente,\\
	Plus rien il n'en restera.\\
	-- \emph{La Marmotte} est épatante,\\
	La Marmotte le fera.
}

\couplet{%
	Le local, quand je l'inspecte,\\
	Exhale une odeur de rat…\\
	Sa propreté m'est suspecte,\\
	Qui donc me le balaiera ?\\
	-- \emph{L'Abeille}, qui se respecte,\\
	L'Abeille vous le fera.
}

\couplet{%
	Il faudrait caser des places\\
	Pour notre prochain gala ;\\
	Le public reste de glace :\\
	Qui donc l'intéressera ?\\
	\emph{L'Hirondelle} se déplace,\\
	L'Hirondelle le fera.
}

\couplet{%
	Ce paquet de vieilles hardes\\
	Pour un pauvre en son grabat :\\
	Il vit dans une mansarde,\\
	Qui me le découvrira ?\\
	-- Puisqu'il s'agit de mansarde,\\
	\emph{Le Hibou} s'en chargera.
}

\couplet{%
	Devant vous je m'agenouille,\\
	Sûr qu'on vous canonis'ra…\\
	Mes patrouilles, mes patrouilles,\\
	Qui vous a changées comm' ça ?…\\
	\emph{-- C'est que toutes vos patrouilles}\\
	\emph{Vous ont vu, Chef, \textsc{fair' comm' ça.}}
}
