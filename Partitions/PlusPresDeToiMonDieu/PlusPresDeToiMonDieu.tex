\couplet{
	J'espère en toi, mon Dieu,\\
	J'espère en toi,\\
	Miséricordieux,\\
	Sois-le pour moi.\\
	Du doute ou de l'effroi\\
	Quand passe le vent froid,\\
	J'espère en toi, mon Dieu,\\
	J'espère en toi.
}

\couplet{
	N'aimer que toi, mon Dieu,\\
	N'aimer que toi,\\
	Tes Saints l'ont su faire, eux,\\
	Pourquoi pas moi ?\\
	En regardant la Croix,\\
	Ô mon Maître, apprends-moi\\
	Comment n'aimer, mon Dieu,\\
	N'aimer que toi.
}

\couplet{
	Plus près de toi, mon Dieu,\\
	Plus près de toi,\\
	C'est là mon humble vœu,\\
	Veux-tu de moi ?\\
	Je voudrais chaque jour\\
	Monter dans ton amour ;\\
	Plus près de toi, mon Dieu,\\
	Plus près de toi.
}

{\footnotesize \emph{Au feu de camp}}

\couplet{
	Conserve-nous, mon Dieu,\\
	Tout près de toi,\\
	Nous qui devant ce feu\\
	Pensons à toi.\\
	Garde notre sommeil,\\
	Que l'heure du réveil\\
	Nous trouve tous, mon Dieu,\\
	Tout près de toi.
}
