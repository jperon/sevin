\couplet{%
	Qu'importe où soit le camp ?\\
	La nature est si belle,\\
	Splendeur qui se révèle\\
	Plus grande à chaque instant :\\
	Qu'importe où soit le camp ?
}

\couplet{%
	Du nord jusqu'au Midi\\
	Notre France est si belle,\\
	Si douce et maternelle\\
	Au cœur de tous ses fils\\
	Du Nord et du Midi.
}

\couplet{%
	Et c'est un tel bonheur\\
	Les âmes sont si belles,\\
	Si claires, les prunelles,\\
	Si forts, les jeunes cœurs ;\\
	Et c'est un tel bonheur.
}

\couplet{%
	Dieu de nos camps d'été,\\
	Que ta main nous protège\\
	Des flots ou de la neige\\
	Au piège redouté,\\
	Bénis nos camps d'été.
}

\couplet{%
	Bénis nos camps d'été,\\
	Garde-les sans reproche,\\
	De toi qu'ils nous rapprochent,\\
	Jusqu'à la sainteté ;\\
	Bénis nos camps d'été.
}
