\couplet{
	En cercle, par Sizaines,\\*
	Tous les Louveteaux\\
	Avecque leurs Cheftaines\\*
	Lancent aux échos :\\
	\textit{Di-ya, \textsc{Akéla est là}, la la la,}\\*
	\textit{Ya la la la la, di-ya la la ;}\\
	\textit{Di-ya, \textsc{Akéla est là}, la la la,}\\*
	\textsc{Nous saluons Akéla !}
}


\couplet{
	Quand jaillissent les gerbes\\*
	De leurs feux de camp\\
	Les Scouts forts et superbes\\*
	Dansent en chantant :\\
	\textit{Di-ya, \textsc{nous voilà !}, la la la,}\\*
	\textit{Ya la la la la, di-ya la la ;}\\
	\textit{Di-ya, \textsc{nous voilà !}, la la la,}\\*
	\textsc{Scouts de France, nous voilà !}
}



\couplet{
	Nos jeunes sœurs les Guides,\\*
	D'un pareil entrain\\
	Reprennent, intrépides,\\*
	Le même refrain :\\
	\textit{Di-ya, \textsc{nous voilà !}, la la la,}\\*
	\textit{Ya la la la la, di-ya la la ;}\\
	\textit{Di-ya, \textsc{nous voilà !}, la la la,}\\*
	\textsc{C'est les Guides que voilà !}
}



\couplet{
	Éparses dans les villes\\*
	Ou par les sentiers,\\
	Voici les voix viriles\\*
	De nos grands Routiers :\\
	\textit{Di-ya, \textsc{nous voilà !}, la la la,}\\*
	\textit{Ya la la la la, di-ya la la ;}\\
	\textit{Di-ya, \textsc{nous voilà !}, la la la,}\\*
	\textsc{C'est la Route que voilà !}
}



\couplet{
	Pour conserver la flamme\\*
	Qui brille en vos yeux,\\*
	Amis, gardez votre âme\\*
	Tout près du Bon Dieu :\\*
	\textit{Di-ya, \textsc{et voilà !}, la la la,}\\*
	\textit{Ya la la la la, di-ya la la ;}\\*
	\textit{Di-ya, \textsc{et voilà !}, la la la,}\\*
	\textsc{Notre secret, le voilà !}
}
