\couplet{%
	Pleins de mystère,\\*
	Tous les buissons\\
	Portent de terre\\
	Au ciel vos chansons.\\
	Près de la flamme\\
	En vos burnous,\\
	De toute votre âme,\\*
	Chantez, mes Loups !
}

\couplet{%
	Loin de vos mères\\*
	Mais près de Dieu,\\
	Ô petits frères,\\
	Sous le grand ciel bleu,\\
	La Vierge Sainte\\
	Veille sur vous ;\\
	Dormez sans crainte,\\*
	Dormez, mes Loups !
}
