\strophe{%
	C'est le Rocher du Conseil,\\*
	Où les vieux-loups nous appellent ;\\*
	Allons, montrons notre zèle.\\*
	C'est le Rocher du Conseil,\\*
	Le Rocher, le Rocher du Conseil.\\
	Les sizeniers aux aguets\\*
	Nous font vite rassembler,\\*
	Écoutez-nous hurler :\\
	C'est le Rocher du Conseil,\\*
	Où les vieux-loups nous appellent ;\\*
	Allons, montrons notre zèle.\\*
	C'est le Rocher du Conseil,\\*
	Le Rocher, le Rocher du Conseil.
}
