\couplet{%
	Il est à moi quand je le prie,\\
	Pourrais-je le quitter ?\\
	Je passerais toute ma vie,\\
	Seigneur, à t'écouter !
}

\couplet{%
	Il est à moi dans tous mes frères,\\
	Caché, mais lumineux ;\\
	En eux, c'est lui que je vénère,\\
	C'est lui que j'aime en eux.
}

\couplet{%
	Il est à moi dans cette Hostie\\
	Qui vient jusqu'en mon cœur ;\\
	Toute mon âme en lui blottie\\
	Tressaille de bonheur !
}

\couplet{%
	Quand la souffrance entre en mon âme,\\
	J'entends sa douce voix\\
	Me murmurer ces mots de flamme :\\
	« À deux portons ma croix ! »
}

\couplet{%
	Quand triste et seul, en proie au doute,\\
	Mon cœur sombre à demi,\\
	J'entends sonner sur la grand'route\\
	Les pas de mon Ami.
}

\couplet{%
	Si le péché frappe à ma porte,\\
	J'appelle à mon secours ;\\
	Sa grâce alors est la plus forte :\\
	Jésus ! sois là toujours !
}

\couplet{%
	Quand de la mort le grand passage\\
	À lui viendra m'unir,\\
	J'espère voir son clair visage\\
	M'aider à bien mourir !
}
