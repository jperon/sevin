\couplet{%
    De la sainte Règle\\*
    Reçois, mon fils, l'enseignement ;\\
    Monte comme l'aigle\\*
    Vers Dieu qui t'a fait son enfant.
}

\refrain{
	Ne tourne pas la tête,\\*
	Qu'en tout Dieu soit glorifié ! \vbis
}

\couplet{%
    Dans l'obéissance\\*
    Servons le Seigneur, notre Roi ;\\
    La concupiscence\\*
    Fuira au doux son de sa voix !
}

\couplet{%
    Le démon te tente ?\\*
    Le Christ brisera ses assauts !\\
    Si le bien t'enchante,\\*
    C'est lui qui te porte si haut.
}

\couplet{%
    À la pénitence\\*
    Jésus, doucement, te convie,\\
    Car, dans sa patience,\\*
    Il veut te mener à la vie.
}

\couplet{%
    Si l'Abbé commande,\\*
    Crois que Dieu te parle par lui ;\\
    Et que ton offrande\\*
    Par lui remonte à Jésus-Christ !
}

\couplet{%
    Aime le silence :\\*
    Qui trop parle, souvent dévie.\\
    De Dieu la présence\\*
    À celui qui l'aime suffit.
}

\couplet{%
    De l'humilité\\*
    Gravis constamment les degrés,\\
    Et la charité\\*
    Croîtra en ton cœur purifié.
}

\couplet{%
    Fidèlement chante\\*
    De l'Église à Dieu les louanges :\\
    Prière constante\\*
    Qu'accompagnent les chœurs des anges.
}

\couplet{%
    La vie éternelle\\*
    Attend qui sert bien son prochain ;\\
    Que, dans ton bon zèle,\\*
    Au Christ tu ne préfères rien !
}
