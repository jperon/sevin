\couplet{
	Le roi Arthur avait trois fils, \emph{quel supplice !}\\*
	Mais c’était un excellent roi, \emph{oui, ma foi !}\\*
	Par lui, ses fils furent chassés, oui chassés à coups de pieds,\\*
	Pour n’avoir pas voulu chanter.
}

\refrain{
	Pour n’avoir pas voulu chanter, ohé !\vbis\\*
	Par lui, ses fils furent chassés, oui chassés à coups de pieds,\\*
	Pour n’avoir pas voulu chanter
}

\couplet{
	Le premier fils se fit tisserand, \emph{oui vraiment !}\\*
	Le second fils se fit meunier, \emph{c’est bien vrai !}\\*
	Le troisième se fit commis d’un tailleur de son pays,\\*
	Un rouleau de drap sous son bras.
}

\couplet{
	Le premier fils volait du fil, \emph{c’est bien vil !}\\*
	Le second fils volait du blé, \emph{c’est bien laid !}\\*
	Et le commis du tailleur volait du drap à toute heure,\\*
	Pour en habiller ses deux sœurs.
}

\couplet{
	Dans l’écluse, le meunier s’noya, \emph{Ah ! ah ! a !}\\*
	À son fil, le tisserand s’pendit. \emph{Hi ! hi ! hi !}\\*
	Et le diable, mis en furie, emporta le p’tit commis,\\*
	Un rouleau de drap sous son bras.
}
