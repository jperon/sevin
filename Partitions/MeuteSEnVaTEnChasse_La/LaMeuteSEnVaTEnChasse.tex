}

\couplet{%
	Le louv'teau pense aux autres,\\*
	Avant d'penser à lui.%
}

\couplet{%
	Bien grands ses yeux il ouvre,\\*
	Et ses oreilles aussi.%
}

\couplet{%
	Louv'teau est toujours propre,\\*
	Et reluit comme un sou.%
}

\couplet{%
	Très loyal et sincère,\\*
	Louv'teau ne ment jamais.%
}

\couplet{%
	Il a l'sourire aux lèvres,\\*
	Et chant' comme un pinson.
}
