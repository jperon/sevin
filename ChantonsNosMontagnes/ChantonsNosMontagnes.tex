\couplet{
	Blanches Pyrénées,
	
	Alpe au front neigeux,
	
	Tout illuminées
	
	Des splendeurs des cieux.
}

\couplet{
	Ô Vosges fidèles,
	
	Vous avez nos cœurs,
	
	Fières citadelles
	
	De nos défenseurs.
}

\couplet{
	Ô Puys de l'Auvergne,
	
	Donjon du Pays,
	
	Sommets des Cévennes,
	
	Soyez tous bénis.
}

\couplet{
	Montagnes puissantes,
	
	Par vous l'on prie mieux,
	
	Et gravir vos pentes,
	
	C'est monter vers Dieu.
}
