\couplet{
	L'un étudiait le droit,
	
	L'autre la méd'cine,
	
	Pour rendre service de-ci,
	
	Pour rendre service de-là,
	
	Pour rendre service.
}

\couplet{
	L'troisème était ouvrier
	
	De ses mains habile
	
	À rendre service de-ci,
	
	À rendre service de-là,
	
	À rendre service.
}

\couplet{
	Partir'nt vers un lotiss'ment
	
	De la banlieue rouge,
	
	Tenter l'aventure de-ci,
	
	Tenter l'aventure de-là,
	
	Tenter l'aventure.
}

\couplet{
	Les voyant passer en scouts,
	
	Chacun s'mit aux portes
	
	Pour voir leur figure de-ci,
	
	Pour voir leur figure de-là,
	
	Pour voir leur figure.%
}

\couplet{%
	\mbox{Comme un chariot s'embourbait}
	
	Dans un' fondrière,
	
	Le sortir'nt de l'ornièr' de-ci,
	
	Le sortir'nt de l'ornièr' de-là,
	
	Le sortir'nt de l'ornière.
}

\couplet{
	\mbox{Dressèr'nt leur tente en un pré ;}
	
	Cent goss's accourrurent.
	
	Les garçons accourrur'nt de-ci,
	
	Les fillett's accourrur'nt de-là,
	
	Cent goss's accourrurent.
}

\couplet{
	Ce petit monde aussitôt
	
	Fut mis en patrouille :
	
	Les garçons en patrouill' de-ci,
	
	Les garçons en patrouill' de-là,
	
	Fur'nt mis en patrouille.
}

\couplet{
	Ils n'arrêtèrent les jeux
	
	Que pour fair' la soupe,
	
	Et donnèr'nt de la soup' de-ci,
	
	Et donnèr'nt de la soup' de-là,
	
	Donnèr'nt de leur soupe.
}

\couplet{
	L'méd'cin soigna les bobos
	
	De tout's les familles :
	
	L'mal de dents des papas de-ci,
	
	Les gros rhum's des mamans \mbox{de-là,}
	
	De tout's les familles.
}

\couplet{
	C'lui qu'était presque avocat
	
	Éclaircit leurs doutes,
	
	À propos d'un loyer de-ci,
	
	À propos d'un mariag' de-là,
	
	Éclaircit leurs doutes.
}

\couplet{
	L'ouvrier faisait jouer
	
	Ses outils sans trève,
	
	Réparait les portes de-ci,
	
	Réparait les f'nêtres de-là,
	
	Réparait sans trève.
}

\couplet{
	Puis enfin, le soir tombé,
	
	Firent un feu d'herbes,
	
	Et chantèr'nt aux lotis de-ci,
	
	Et chantèr'nt aux lotis de-là,
	
	Leurs chansons superbes.
}

\couplet{
	Quand le feu fut presque mort,
	
	Dirent leur prière,
	
	Et pour la premièr' fois de-ci,
	
	Et pour la premièr' fois de-là,
	
	Les gens se signèrent.
}

\couplet{
	S'en allèr'nt, serrant les mais
	
	Rudes comme chênes,
	
	Et tout l'mond' leur disait de-ci,
	
	Et tout l'mond' leur disait de-là,
	
	« À la s'main' prochaine ! »
}
