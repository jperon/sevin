\couplet{
	Vierge si douce aux yeux limpides,\\*
	Tourne tes regards vers tes Guides,\footnotemark\\*
	Alleluia ; \emph{(bis)}\\
	Conduis-nous par tes sentiers sûrs\\*
	Pour que nos yeux demeurent purs,\\*
	O Maria,\\
	Alleluia.
}

\couplet{
	Toi qui voulus nous rassembler,\\*
	Oh ! ne nous laisse plus trembler,\\*
	Alleluia ; \emph{(bis)}\\
	Chasse à jamais de notre cœur\\*
	La défiance et la rancœur,\\*
	O Maria,\\*
	Alleluia.
}

\couplet{
	Fais-nous à tous toujours sourire,\\*
	Comme toi, souffrir sans le dire,\\*
	Alleluia ; \emph{(bis)}\\
	Et panser de nos doigts très doux\\*
	Ceux qui souffrent autour de nous,\\*
	O Maria,\\*
	Alleluia.
}

\couplet{
	Si petites que nous soyons,\footnotemark\\*
	Fais-nous répandre les rayons,\\*
	Alleluia ; \emph{(bis)}\\
	Les rayons d'or de ta bonté,\\*
	Plus douce encor que ta beauté,\\*
	O Maria,\\*
	Alleluia.
}

\couplet{
	Si jamais, corps et cœur blessés,\\*
	Nous n'avons plus de force assez,\\*
	Alleluia ; \emph{(bis)}\\
	Soutiens-nous, Mère, jusqu'au bout,\\*
	Pour que nous chantions malgré tout,\\*
	O Maria,\\*
	Alleluia.
}

\couplet{
	Quand nos yeux verront se lever\\*
	Le jour des jours longtemps rêvé,\\*
	Alleluia ; \emph{(bis)}\\
	Montre-nous dans tes bras qui ploient\\*
	Jésus, notre éternelle Joie,\\*
	O Maria,\\*
	Alleluia.
}
